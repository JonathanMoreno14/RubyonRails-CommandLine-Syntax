  a_string = "Hello World"  # A string with double qoutes

######Interpolation
```ruby

  h ="hello" 
  
  w = "World"
  
  "#{h} #{w}" #Prints out hello World
  
  puts("The program says " + "#{h} #{w}")  # Prints out The program says hello World

```
